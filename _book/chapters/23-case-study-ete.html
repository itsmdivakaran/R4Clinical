34  End-to-End Clinical Trial Analysis

34.1 Introduction: The Complete Clinical Trial Pipeline

A clinical trial generates terabytes of data from dozens of sources: enrollment forms, vital signs monitors, laboratory instruments, adverse event reports, and pharmacy records. Transforming this raw chaos into a regulatory-quality submission requires a rigorous, documented pipeline that bridges study design, data collection, standardisation, analysis, and reporting.

This chapter walks through a Phase III randomised controlled trial for hypertension — a common, well-understood indication that showcases the full spectrum of clinical data challenges: handling missing data, SDTM standardisation, ADaM derivations, endpoint analysis, safety tabulations, and regulatory-grade quality control.

Learning objectives by chapter end: 1. Map raw clinical data to SDTM domains using pharmaversesdtm 2. Build analysis-ready datasets (ADaM) using admiral-style derivations 3. Execute primary and secondary endpoint analyses 4. Generate safety tables and figures 5. Implement QC procedures aligned with FDA/EMA expectations 6. Understand the regulatory landscape (E9, CDISC, CDASH)

34.3 Step 2: Raw Data to SDTM — Pharmaversesdtm Domains

SDTM (Study Data Tabulation Model) is the FDA-mandated standard for clinical trial data. pharmaversesdtm provides example data in SDTM format across six domains critical for a hypertension trial:

  1. DM — Demographics
  2. VS — Vital Signs (blood pressure, heart rate)
  3. LB — Laboratory Results (electrolytes, renal function)
  4. AE — Adverse Events
  5. EX — Exposure (dosing records)
  6. EC — Findings (visit dates, randomisation flags)

34.4 Step 3: SDTM to ADaM — Building Analysis-Ready Datasets

ADaM (Analysis Data Model) datasets are analysis-ready, with variables derived and preprocessed. We build two key ADaM datasets:

  1. ADSL (Subject-Level) — demographics, baseline, randomisation, population flags
  2. ADVS (Vital Signs Analysis) — BP with baseline, change, and visit-level analysis flags

34.4.1 3.1 Build ADSL (Subject-Level Analysis Dataset)

# Start with DM as the foundation
adsl <- dm %>%
  filter(!is.na(ARM), ARM != "", ARM != "Screen Failure") %>%
  select(USUBJID, ARMCD, ARM, AGE, SEX, RACE) %>%
  mutate(
    # Map actual ARM values to study treatment labels
    TRT01P = factor(
      case_when(ARM == "Placebo" ~ "Placebo", TRUE ~ "Drug XYZ"),
      levels = c("Placebo", "Drug XYZ")
    ),
    TRT01PN = as.numeric(TRT01P),

    # Randomisation date (derive from early VS/LB records)
    RANDDT = as.Date("2023-01-15"),  # For this example, fixed date

    # Demographics flags
    SAFFL = "Y",  # Safety population = all randomised
    ITTFL = "Y",  # ITT population = all randomised
    PPFL = ifelse(ARMCD %in% c("Active", "Placebo"), "Y", "N"),  # Per-protocol: exclude early withdrawals

    # Age-group for subgroup analysis
    AGEGRP = case_when(
      AGE < 45 ~ "<45",
      AGE < 60 ~ "45–60",
      AGE < 75 ~ "60–75",
      TRUE ~ "≥75"
    ),

    # Weight/BMI categories (if available; otherwise use age/sex)
    SEXF = ifelse(SEX == "F", "Female", "Male")
  )

# Merge baseline SBP and DBP (earliest record per subject/test)
baseline_vitals <- vs_bp %>%
  filter(VSTESTCD %in% c("SYSBP", "DIABP"), !is.na(VSVAL)) %>%
  group_by(USUBJID, VSTESTCD) %>%
  slice_min(VSDT, n = 1, with_ties = FALSE) %>%
  ungroup() %>%
  select(USUBJID, VSTESTCD, VSVAL) %>%
  pivot_wider(
    id_cols = USUBJID,
    names_from = VSTESTCD,
    values_from = VSVAL
  ) %>%
  rename(BLSBP = SYSBP, BLDBP = DIABP)

adsl <- adsl %>%
  left_join(baseline_vitals, by = "USUBJID") %>%
  mutate(
    # Baseline categories for stratification
    BLSBPGRP = ifelse(BLSBP >= 160, "≥160", "<160"),
    BLDBPGRP = ifelse(BLDBP >= 100, "≥100", "<100")
  )

# Merge baseline lab values (potassium, creatinine)
baseline_labs <- lb %>%
  filter(LBTESTCD %in% c("K", "CREAT")) %>%
  { if ("LBBLFL" %in% names(.) && any(.[["LBBLFL"]] == "Y", na.rm = TRUE))
      filter(., LBBLFL == "Y")
    else
      group_by(., USUBJID, LBTESTCD) %>%
        slice_min(LBDY, n = 1, with_ties = FALSE) %>%
        ungroup() } %>%
  select(USUBJID, LBTESTCD, LBSTRESN) %>%
  pivot_wider(
    id_cols = USUBJID,
    names_from = LBTESTCD,
    values_from = LBSTRESN,
    names_prefix = "BL_"
  )

adsl <- adsl %>%
  left_join(baseline_labs, by = "USUBJID") %>%
  rename(any_of(c(BLPOTASSIUM = "BL_K", BLCREATININE = "BL_CREAT")))

# Count baseline safety events/exclusions for documentation
adsl <- adsl %>%
  mutate(
    # Population assignment (derived from visit attendance and protocol adherence)
    COMPLFL = "Y",  # Assume all completed 12 weeks for now
    EFFICACY_POP = ITTFL
  ) %>%
  select(
    USUBJID, ARMCD, ARM, TRT01P, TRT01PN, AGE, AGEGRP, SEX, SEXF, RACE,
    RANDDT, SAFFL, ITTFL, PPFL, COMPLFL, EFFICACY_POP,
    BLSBP, BLDBP, BLSBPGRP, BLDBPGRP,
    any_of(c("BLPOTASSIUM", "BLCREATININE"))
  )

## Step 3: SDTM to ADaM — Building Analysis-Ready Datasets

ADaM (Analysis Data Model) datasets are analysis-ready, with variables derived and preprocessed. We build two key ADaM datasets:

1. **ADSL** (Subject-Level) — demographics, baseline, randomisation, population flags
2. **ADVS** (Vital Signs Analysis) — BP with baseline, change, and visit-level analysis flags

### 3.1 Build ADSL (Subject-Level Analysis Dataset)

```{r}
#| label: create-adsl
#| message: false

# Start with DM as the foundation
adsl <- dm %>%
  filter(!is.na(ARM), ARM != "", ARM != "Screen Failure") %>%
  select(USUBJID, ARMCD, ARM, AGE, SEX, RACE) %>%
  mutate(
    # Map actual ARM values to study treatment labels
    TRT01P = factor(
      case_when(ARM == "Placebo" ~ "Placebo", TRUE ~ "Drug XYZ"),
      levels = c("Placebo", "Drug XYZ")
    ),
    TRT01PN = as.numeric(TRT01P),

    # Randomisation date (derive from early VS/LB records)
    RANDDT = as.Date("2023-01-15"),  # For this example, fixed date

    # Demographics flags
    SAFFL = "Y",  # Safety population = all randomised
    ITTFL = "Y",  # ITT population = all randomised
    PPFL = ifelse(ARMCD %in% c("Active", "Placebo"), "Y", "N"),  # Per-protocol: exclude early withdrawals

    # Age-group for subgroup analysis
    AGEGRP = case_when(
      AGE < 45 ~ "<45",
      AGE < 60 ~ "45–60",
      AGE < 75 ~ "60–75",
      TRUE ~ "≥75"
    ),

    # Weight/BMI categories (if available; otherwise use age/sex)
    SEXF = ifelse(SEX == "F", "Female", "Male")
  )

# Merge baseline SBP and DBP (earliest record per subject/test)
baseline_vitals <- vs_bp %>%
  filter(VSTESTCD %in% c("SYSBP", "DIABP"), !is.na(VSVAL)) %>%
  group_by(USUBJID, VSTESTCD) %>%
  slice_min(VSDT, n = 1, with_ties = FALSE) %>%
  ungroup() %>%
  select(USUBJID, VSTESTCD, VSVAL) %>%
  pivot_wider(
    id_cols = USUBJID,
    names_from = VSTESTCD,
    values_from = VSVAL
  ) %>%
  rename(BLSBP = SYSBP, BLDBP = DIABP)

adsl <- adsl %>%
  left_join(baseline_vitals, by = "USUBJID") %>%
  mutate(
    # Baseline categories for stratification
    BLSBPGRP = ifelse(BLSBP >= 160, "≥160", "<160"),
    BLDBPGRP = ifelse(BLDBP >= 100, "≥100", "<100")
  )

# Merge baseline lab values (potassium, creatinine)
baseline_labs <- lb %>%
  filter(LBTESTCD %in% c("K", "CREAT")) %>%
  { if ("LBBLFL" %in% names(.) && any(.[["LBBLFL"]] == "Y", na.rm = TRUE))
      filter(., LBBLFL == "Y")
    else
      group_by(., USUBJID, LBTESTCD) %>%
        slice_min(LBDY, n = 1, with_ties = FALSE) %>%
        ungroup() } %>%
  select(USUBJID, LBTESTCD, LBSTRESN) %>%
  pivot_wider(
    id_cols = USUBJID,
    names_from = LBTESTCD,
    values_from = LBSTRESN,
    names_prefix = "BL_"
  )

adsl <- adsl %>%
  left_join(baseline_labs, by = "USUBJID") %>%
  rename(any_of(c(BLPOTASSIUM = "BL_K", BLCREATININE = "BL_CREAT")))

# Count baseline safety events/exclusions for documentation
adsl <- adsl %>%
  mutate(
    # Population assignment (derived from visit attendance and protocol adherence)
    COMPLFL = "Y",  # Assume all completed 12 weeks for now
    EFFICACY_POP = ITTFL
  ) %>%
  select(
    USUBJID, ARMCD, ARM, TRT01P, TRT01PN, AGE, AGEGRP, SEX, SEXF, RACE,
    RANDDT, SAFFL, ITTFL, PPFL, COMPLFL, EFFICACY_POP,
    BLSBP, BLDBP, BLSBPGRP, BLDBPGRP,
    any_of(c("BLPOTASSIUM", "BLCREATININE"))
  )

cat("ADSL Dataset Created:\n")
cat("  Dimensions:", nrow(adsl), "×", ncol(adsl), "\n")
cat("  First 5 records:\n")
print(head(adsl, 5))

cat("\nADSL Summary Statistics:\n")
cat("  Treatment group distribution:\n")
print(table(adsl$ARM))
cat("  Population flags:\n")
print(table(adsl$EFFICACY_POP))
```

### 4.2 ANCOVA Analysis (Primary Endpoint)

```{r}
#| label: ancova-primary-ep
#| message: false
library(emmeans)

# ANCOVA: Change in SBP ~ Treatment + Baseline SBP
ancova_model <- lm(CHG_SBP_ANAL ~ TRT01P + BLSBP, data = primary_data)

cat("ANCOVA Model Summary:\n")
print(summary(ancova_model))

# Extract adjusted means and treatment difference
ancova_emmeans <- emmeans(ancova_model, specs = "TRT01P",
                           at = list(BLSBP = mean(primary_data$BLSBP, na.rm = TRUE)))

cat("\n\nAdjusted Mean Change in SBP (ANCOVA):\n")
print(ancova_emmeans)

# Treatment contrast (Drug vs Placebo)
contrast_result <- contrast(ancova_emmeans, method = "pairwise")

cat("\n\nTreatment Difference (Drug XYZ vs Placebo):\n")

tx_diff <- contrast_df$estimate[1]
tx_se <- contrast_df$SE[1]
tx_ci_lower <- tx_diff - 1.96 * tx_se
tx_ci_upper <- tx_diff + 1.96 * tx_se
p_value <- contrast_df$p.value[1]

cat("\n\nPrimary Endpoint Result:\n")
cat("Treatment Difference (Drug XYZ vs Placebo):\n")
```

```{r}
log_model <- glm(RESPONDER ~ TRT01P,
                 data = primary_data %>%
                   mutate(RESPONDER = ifelse(CHG_SBP_ANAL <= -20, 1, 0)),
                 family = binomial(link = "logit"))

# Extract OR
log_coef  <- coef(log_model)
log_se    <- summary(log_model)$coef[, "Std. Error"]
trt_terms <- setdiff(names(log_coef), "(Intercept)")
if (length(trt_terms) > 0) {

    ci_upper <- exp(log_coef[term] + 1.96 * log_se[term])
```
